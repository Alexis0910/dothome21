26. 함수 유형 : 객체 생성자 함수

{
    function func(num, name, job) {
        this.num = num;
        this.name = name;
        this.job = job;
        
        this.result = function(){
            document.write(this.num + ". 내 이름은 "+ this.name +"이며, 직업은 " + this.job + "이다.");
        }
    }

    const info1 = new func("1", "소은", "웹퍼블리셔");   //인스턴스 생성
    const info2 = new func("2", "Alexis", "프론트엔드 개발자");   //인스턴스 생성

    info1.result();
    info2.result();
}

2. 내 이름은 Alexis이며, 직업은 프론트엔드 개발자이다.
