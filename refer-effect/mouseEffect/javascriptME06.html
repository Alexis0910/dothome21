const cursor = document.querySelector(".cursor");
const width = cursor.clientWidth;
const height = cursor.clientHeight;
const circle = cursor.getBoundingClientRect();

function moveCircle(e){
    gsap.to(cursor, {duration: 0.3, left: e.pageX - circle.width/2, top: e.pageY - circle.height/2});
    
    //출력
    document.querySelector(".pageX").innerText = pageX;
    document.querySelector(".pageY").innerText = pageY;

};
window.addEventListener("mousemove", moveCircle);
