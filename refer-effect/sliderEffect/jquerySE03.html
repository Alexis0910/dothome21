const sliderWrap = $(".slider-wrap");
const sliderImg = $(".slider-img");
const sliderInner = $(".slider-inner");
const slider = $(".slider");

let currentIndex = 0;
let sliderWidth = slider.width();

//콜백함수 (함수안의 함수 1 안의 2 이런식 다른함수)
//재귀함수(함수안의 같은함수)
setInterval(function(){
    sliderInner.animate({ left: -sliderWidth }, 400, function(){
        $(".slider:first-child").appendTo(sliderInner);
        //슬라이더의 첫번째(eq는진짜 찐순서, first-child는 바뀜)를 sliderInner의 맨뒤로보낸다 
        //.append() : $(A).append(B) A 사이의 마지막에 B 가 추가 된다.
        //appendTo() : $(A).appendTo(B) A 가 B 사이의 마지막에 추가 된다.

        sliderInner.css("left","");
        //첫번째이미지를 맨뒤로보내면 left-800도 0으로 만들어서 0, -800(한장넘기기) , 0, -800(한장넘기기) 반복해서 옆으로 넘기는것
    });
}, 3000);
