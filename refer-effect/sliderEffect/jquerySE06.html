const sliderWrap = $(".slider-wrap");
const sliderImg = $(".slider-img");
const sliderInner = $(".slider-inner");
const slider = $(".slider");
const sliderBtn = $(".slider-btn");
const sliderDot = $(".slider-dot");

let sliderCount = slider.length;
let sliderWidth = slider.width();
let sliderTotalWidth = sliderWidth * sliderCount;
let currentIndex = 0;
let dotIndex = "";

//이미지가 100장들어가면 80000px되니까 css에서 width값은 이제 무의미해서 지움
sliderInner.css({ width: sliderTotalWidth});
slider.css({ width: sliderWidth });

//dot버튼 생성하기
//이미지 갯수만큼 닷 메뉴 생성
slider.each(function(index){
    dotIndex += ""+ index +"";
});
sliderDot.html(dotIndex);        
//sliderDot안의 a태그를 찾아서 eq(순서 currentIndex는 0이니까 첫번째)에 active추가
sliderDot.find("a").eq(currentIndex).addClass("active")

//이미지 움직이기
function gotoSlider(num){
    //gotoslider안에 prev, next 애니메이션, 닷메뉴actvie둘다 들어감
    sliderInner.animate({ left: -sliderWidth * num}, 600, "easeInQuart");
    //currentIndex를 ++주기만 하는게아니라 그 전값인 num을 대입시켜줘서 인덱스값을 새로 update시킴
    currentIndex = num;
    //console.log(currentIndex)
    //닷메뉴에 active, remove하는거
    sliderDot.find("a").removeClass("active").eq(currentIndex).addClass("active")
};

//dot버튼 클릭하면 이동하기 3번 클릭하면 이미지3 나오게
// dot의 index도 01234,sliderInner의 num도 01234 * 800px, 일치하는거지 ㅇㅇ!!!!!
sliderDot.on("click", "a", function(e){
    e.preventDefault();
    
    gotoSlider($(this).index());
});

//버튼 클릭하기 on사용시 : Btn안의 a태그 click이벤트
// sliderBtn.click(function(){});
sliderBtn.on("click", "a", function(e){
    e.preventDefault();
    let nextIndex = (currentIndex + 1) % sliderCount;
    let prevIndex = (currentIndex - 1) % sliderCount;
    if(currentIndex == 0) prevIndex = sliderCount -1;
    //만약 현재인덱스가 0일때(이미지1번일때) prevIndex는 
    //console.log(prevIndex)

    if( $(this).hasClass("prev")) {
        gotoSlider(prevIndex);
    } else {
        gotoSlider(nextIndex);
    }
});
