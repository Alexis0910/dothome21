$(".paraDot ul li").click(function(e){
    e.preventDefault();
    let target = $(this); //타겟설정 : 사용자가 클릭한 버튼을 저장
    let index = target.index() //인덱스(순서)설정 몇번째버튼을 클릭했는지 알아내는것
    let section = $(".content-item").eq(index); //6번 클릭하면 6번섹션에 저장된다.
    
    let offset = section.offset().top; //각섹션의 offset값 알아내기.
    $("html, body").animate({scrollTop: offset}, 600, "easeInExpo"); //6초동안 스크롤이 부드럽게움직임
});

$(window).scroll(function(){
    let scrollTop = $(this).scrollTop(); //같은window라 this개능

    $(".paraScroll span").text(parseInt(scrollTop));
    
    //for문
    for(let i=0; i<$(".content-item").length; i++ ) {
        if(scrollTop >= $(".content-item").eq(i).offset().top -2) {
            $(".paraDot ul li").eq(i).addClass("active").siblings().removeClass("active");
        }
    }
    //each문

    //    }

});
