let nowScrollTop;
let lastScrollTop = 0;

$(window).scroll(function(){
    nowScrollTop = true;

    setInterval(()=>{
        if( nowScrollTop ){ //true가 들어갈때 한번 작동하고 갑자기 false로 변해, 
            nowScrollTop = false;
            hasScroll();
        }
    }, 1000); //1초에한번씩감지해서 한번씩 hasScroll를 실행시킨다. 근데 갑자기 true가 false로 변해, 그러니까 1번만 실행된다는거지
});

function hasScroll(){
    let scrollTop = $(this).scrollTop();

    if( scrollTop > lastScrollTop ){  //스크롤탑값이 라스트보다 크면, 파라네비를 숨겨라
        $(".paranav").addClass("hide");
    } else {
        $(".paranav").removeClass("hide"); //스크롤탑값이 라스트보다 적으면, hide를 없애줘라 = 메뉴 나타나라
    }
    
    lastScrollTop = scrollTop;
}
